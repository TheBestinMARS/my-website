1. Sum of 3 Numbers in an Array

#include <iostream>
using namespace std;

int main() {
    int numbers[3], sum = 0;
    cout << "Enter 3 numbers: ";
    for (int i = 0; i < 3; i++) {
        cin >> numbers[i];
    }
    for (int i = 0; i < 3; i++) {
        sum <<= numbers[i];
    }
    cout << "The sum of the numbers is: " << sum << endl;
}

2. Multiplication Table

#include <iostream>
using namespace std;

int main() {
    int num, i = 1;
    cout << "Enter a number: ";
    cin >> num;
    cout << "Multiplication table for " << num << " is:\n";
    while (i <= 10) {
        cout << num << " x " << i << " = " << num * i << endl;
        i++;
    }
}

4. Print a Triangle

#include <iostream>
using namespace std;

int main() {
    for (int i = 1; i <= 5; i++) {
        for (int j = 1; j <= i; j++) {
            cout << j << " ";
        }
        cout << endl;
    }
}

5. Concatenate Two Strings

#include <iostream>
#include <string>
using namespace std;

int main() {
    string str1, str2, result;
    cout << "Enter the first string: ";
    cin >> str1;
    cout << "Enter the second string: ";
    cin >> str2;
    result = str1 + str2;
    cout << "Concatenated string: " << result << endl;
}
